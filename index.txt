  <script type="module">
    const appId = "sq0idp-UHFeGtPq7o_wZwrPngY1cQ";        // from Developer Dashboard
    const locationId = "LEADGZ7813668";          // from Developer Dashboard

    const payments = window.Square.payments(appId, locationId);

    async function initialize() {
      try {
        const applePay = await payments.applePay({ countryCode: "US", currencyCode: "USD" });
        await applePay.attach("#apple-pay-button");
      } catch (e) {
        console.log("Apple Pay unavailable", e);
      }

      try {
        const googlePay = await payments.googlePay({ countryCode: "US", currencyCode: "USD" });
        await googlePay.attach("#google-pay-button");
      } catch (e) {
        console.log("Google Pay unavailable", e);
      }
    }

    initialize();
  </script>
